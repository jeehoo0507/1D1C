그림에도 무언가 큰 힌트가 있을 것이라 생각해서 각 숫자 변화에 집중해보았는데 보니 2차부터는 거의 모든 숫자가 나오는거 같았고 0차 1차의 예외 케이스를 따지면 되는 문제 같았다.
2차에 등장하는 숫자들에는 예외가 없어보여 0차 1차 숫자를 하나씩 써보고 안되는 걸 예외 케이스로 넣었다. 처음에 1, 4 만 생각했는데 알고보니 6을 빼먹었다.

차수가 늘 때 모든 숫자가 차수에 등장한다는 가정하에 
하한값 x 2 +1이 다음 차수의 하한이 된다. 이를 이용하면 그냥 끝나는 문제임


```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {

    // 구상 
    // 일단 처음에는 2,3,5,7... 모든 소수
    // 2번째는 5,7,8,9,10,11,...   
    // 3번쨰는 12,13,14,15,...
    // 먼가 다 있는 듯??? 음 처음 이후에도 6뺴고는 다 있음 뭔가 그 소수 존재 확률인가 x/lnx?
    // 알던 무조건 있다고 가정 일단 다 있는 듯 하한만 있고
    //절대 안됨 1,4,6 <-- 이게 문제임

    int n;
    cin >> n;

    
    if (n < 12) {
        if (n == 1 || n == 4 || n == 6) 
        {
            cout << -1 << "\n";
        } 
        else if (n == 2 || n == 3 || n == 11)
        {
            cout << 0 << "\n";
        }
         else if (n == 5 || n == 7 || n == 8 || n == 9 || n == 10)
        {
            cout << 1 << "\n";
        }
        return 0;
    }

    int chasu = 2;
    long long hahan = 12;

    while (1) {
        long long nmin = 2 * hahan + 1;
        if (nmin <= n) 
        {
            hahan = nmin;
            chasu++;
        } 
        else break;
    }

    cout << chasu << "\n";

    return 0;
}
```
