#include <iostream>

#include <set>

#include <vector>

#include <iomanip>

using namespace std;

  

int main(){

long long int n,l;

cin>>n>>l;

set<int> s = {};

  

int c;

while(n--){

cin>>c;

s.insert(c);

}

  

vector<int> v(s.begin(),s.end());

  

int max = 0;

if(v[0]==0){

max = (l - v[v.size()-1]) *2;

for(int i=1;i<v.size();i++){

if(v[i] - v[i-1]>max) max = v[i] - v[i-1];

else;

}

}

else{

int a = v[0]*2;

int b = (l - v[v.size()-1])*2;

  

max = a>b? a:b;

  

for(int i=1;i<v.size();i++){

if(v[i] - v[i-1]>max) max = v[i] - v[i-1];

else;

}

}

cout << fixed << setprecision(10) << (double)max / 2;

  

return 0;

}
