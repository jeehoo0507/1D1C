사전 순 비교는 str1 > str2 가 성립함

대소문자 구분을 없애야함

```cpp title:PetyaandStrings.cpp
#include <iostream>
using namespace std;

char bigtosmall(char n){
    if(n<97) n += 32;
    return n;
}

int main(){
    string str1, str2;
    cin >> str1;
    cin >> str2;
    for(int i=0;i<str1.length();i++){
        str1[i] = bigtosmall(str1[i]);
        str2[i] = bigtosmall(str2[i]);
    }

    if(str1 == str2) cout<<0;
    else if(str1 > str2) cout<<1;
    else cout<<-1;

    return 0;
}
```
