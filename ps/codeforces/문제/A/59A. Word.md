대문자 변환하고 소문자 변환하고 원래하고 비교해서 차이가 적은걸 출력!
```cpp title:Word.cpp
#include <iostream>
#include <cmath>

  

using namespace std;

  

char bigtosmall(char a){

if(a<97) a+=32;

else;

return a;

}

char smalltobig(char b){

if(b>96) b-=32;

else;

return b;

}

  

int addf(string str4){

int result =0;

for(int k=0;k<str4.length();k++){

result += str4[k];

}

return result;

}

  

int main(){

string str;

cin>>str;

  

string str1 = str;

string str2 = str;

  

for(int i=0;i<str1.length();i++){

str1[i] = bigtosmall(str1[i]);

}

  

for(int j=0;j<str2.length();j++){

str2[j] = smalltobig(str2[j]);

}

  

long long int str11 = abs(addf(str)-addf(str1));

long long int str22 = abs(addf(str)-addf(str2));

  

if(str11>str22) cout<<str2;

else cout<<str1;

  

return 0;

}
```
