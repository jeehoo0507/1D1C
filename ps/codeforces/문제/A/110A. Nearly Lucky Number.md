#include <iostream>
using namespace std;

int main(){
    string str1;
    cin>>str1;
    int result = 0;

    for(int i=0;i<str1.length();i++){
        if(str1[i]=='4' || str1[i]=='7') result++;
        else;
    }


    if(result == 4 || result == 7) cout<<"YES";
    else cout<<"NO";

    return 0;
}
