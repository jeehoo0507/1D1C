```cpp title:TheatreSquare.cpp
#include <iostream>
using namespace std;

int main(){
    long long int n,m,a;

    long long int n2,m2;
    
    cin>>n>>m>>a;
    if(n%a==0) n2 = n/a;
    else n2 = n/a+1;

    if(m%a==0) m2 = m/a;
    else m2 = m/a+1;

    cout<<n2*m2;


    return 0;
}
```
