#include <iostream>
using namespace std;

long long int mone(long long int n){
    if(n%10==0) return n/10;
    else return --n;
}


int main(){
    long long int n;
    int k;

    cin>>n>>k;

    for(int i=0;i<k;i++){
        n = mone(n);
    }

    cout<<n;
    return 0;
}
