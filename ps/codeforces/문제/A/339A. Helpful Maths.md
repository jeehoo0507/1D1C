using namespace std;

string str1;

{

num[str1[i]-'0'-1]++;

}

  

int one = num[0];

int two = num[1];

int three = num[2];

  
  

while(one--)

{

cout<<1;

if(one > 0 || two > 0 || three > 0) cout<<'+';

}

  

while(two--)

{

cout<<2;

if(two > 0 || three > 0) cout<<'+';

}

int three2 = three-1;
