#include <iostream>
using namespace std;

int main(){
    int k,n,w;
    cin>>k>>n>>w;
    int result = 0;

    for(int i=1;i<=w;i++){
        result += i*k;
    }

    if(result>n) cout<<result-n;
    else cout<<0;

    return 0;
}
