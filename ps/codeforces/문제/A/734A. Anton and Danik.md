#include <iostream>
using namespace std;

int main(){
    int n;
    cin>>n;
    int Anton = 0;
    int Danik = 0;

    string str1;
    cin>>str1;

    for(int i=0;i<str1.length();i++){
        if(str1[i] == 'A') Anton++;
        else Danik++;
    }

    if(Anton>Danik) cout<<"Anton";
    else if(Anton<Danik) cout<<"Danik";
    else cout<<"Friendship";


    return 0;
}
