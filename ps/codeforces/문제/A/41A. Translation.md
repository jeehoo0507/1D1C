#include <iostream>
using namespace std;

int main(){ 
    string str1;
    string str2;

    cin>>str1;
    cin>> str2;

    int true1 = 1;
    for(int i=0;i<str1.length();i++){
        if(str1[i] == str2[str1.length()-1-i]);
        else true1 = 0;
    }


    cout << (true1? "YES":"NO");

    return 0;
}
