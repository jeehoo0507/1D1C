using namespace std;

  

int main(){

int x;

while(x>0){

if(x>=5){

x-=5;

result++;

}

else if(x>=4){

x-=4;

result++;

}

else if(x>=3){

x-=3;

result++;

}

else if(x>=2){

x-=2;

result++;

}

else if(x>=1){

x-=1;

result++;

}

}

  

cout<<result;

return 0;

}
