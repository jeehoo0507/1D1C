#include <iostream>
using namespace std;

int main(){
    int N;
    cin >> N;

    long long arr[200002] = {};
    long long sum1[200002] = {};  
    long long sum2[200002] = {};  
    long long dp1[200002] = {};   
    long long dp2[200002] = {}; 

    for(int i = 1; i <= N; i++){
        cin >> arr[i];
    }

    int count1 = 0, count2 = 0;
    for(int j = 1; j+2 <= N; j += 2) sum1[++count1] = arr[j] + arr[j+2];
    for(int k = 2; k+2 <= N; k += 2) sum2[++count2] = arr[k] + arr[k+2];


    dp1[1] = max(0LL, sum1[1]);
    dp1[2] = max(dp1[1], sum1[2]);
    dp2[1] = max(0LL, sum2[1]);
    dp2[2] = max(dp2[1], sum2[2]);

    for(int i = 3; i <= count1+1; i++){
        dp1[i] = max(dp1[i-1], dp1[i-2] + sum1[i]);
        dp2[i] = max(dp2[i-1], dp2[i-2] + sum2[i]);
    }

    cout << dp1[count1] + dp2[count2] << "\n";
    return 0;
}
