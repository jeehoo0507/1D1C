ab       % 공백 없음 (기본 붙임)
a\,b     % thin space
a\:b     % medium space
a\;b     % thick space
a\ b     % 백슬래시+스페이스 = 일반 한 칸
a\quad b   % 1em 공백
a\qquad b  % 2em 공백 (\quad의 2배)
