## 학습
```python
import pandas as pd
import lightgbm as lgb
import joblib
from sklearn.model_selection import cross_val_score


TRAIN_PATH = "학습 데이터 경로"
MODEL_PATH = "모델 저장 경로"


def load_data(path):
    df = pd.read_csv(path)
    group_cols = [c for c in df.columns if c.startswith("Group")]
    active_cols = [c for c in group_cols if df[c].sum() > 0]
    X = df[active_cols]
    y = df["Tm"]
    return X, y, active_cols


def build_model():
    return lgb.LGBMRegressor(
        n_estimators=500,
        learning_rate=0.05,
        num_leaves=31,
        random_state=42,
        verbose=-1,
    )


def evaluate(model, X, y):
    r2 = cross_val_score(model, X, y, cv=5, scoring="r2").mean()
    mae = -cross_val_score(model, X, y, cv=5, scoring="neg_mean_absolute_error").mean()
    return r2, mae


def show_importance(model, feature_names, top_n=10):
    ranking = sorted(zip(feature_names, model.feature_importances_),
                     key=lambda pair: pair[1], reverse=True)
    print(f"\n예측에 가장 영향을 준 피처 TOP {top_n}:")
    for name, score in ranking[:top_n]:
        print(f"  {name:12s} {score}")


def main():
    X, y, feature_names = load_data(TRAIN_PATH)
    print(f"분자 {len(X)}개, 피처 {len(feature_names)}개")

    model = build_model()

    r2, mae = evaluate(model, X, y)
    print(f"교차검증 성능 -> R² = {r2:.3f}, MAE = {mae:.2f} K")

    model.fit(X, y)
    print("최종 모델 학습 완료")

    joblib.dump({"model": model, "features": feature_names}, MODEL_PATH)
    print(f"모델 저장 -> {MODEL_PATH}")

    show_importance(model, feature_names)


if __name__ == "__main__":
    main()
```

## 그룹 분석
```python
import pandas as pd
import numpy as np
import joblib
from rdkit import Chem
from rdkit.Chem import Descriptors, Fragments
from rdkit import RDLogger

RDLogger.DisableLog("rdApp.*")

TRAIN_PATH = "학습 데이터 경로"
MODEL_PATH = "모델 저장 경로"
TOP_N = 10


def load_model_and_top_features(model_path, top_n):
    saved = joblib.load(model_path)
    model = saved["model"]
    features = saved["features"]
    ranking = sorted(zip(features, model.feature_importances_),
                     key=lambda pair: pair[1], reverse=True)
    return ranking[:top_n]


def build_candidate_table(smiles_list):
    mols = [Chem.MolFromSmiles(s) for s in smiles_list]
    candidates = {}
    for name, func in Fragments.__dict__.items():
        if name.startswith("fr_"):
            try:
                candidates[name] = [func(m) for m in mols]
            except Exception:
                pass
    candidates["aromatic_atoms"] = [sum(1 for a in m.GetAtoms() if a.GetIsAromatic()) for m in mols]
    candidates["ring_count"] = [Descriptors.RingCount(m) for m in mols]
    candidates["carbon_count"] = [sum(1 for a in m.GetAtoms() if a.GetSymbol() == "C") for m in mols]
    return pd.DataFrame(candidates)


def match_group(group_values, candidate_table):
    best_name, best_r = None, 0.0
    for cand_name in candidate_table.columns:
        cand_values = candidate_table[cand_name].values
        if cand_values.std() == 0:
            continue
        r = np.corrcoef(group_values, cand_values)[0, 1]
        if abs(r) > abs(best_r):
            best_r, best_name = r, cand_name
    exact = (group_values == candidate_table[best_name].values).mean()
    return best_name, best_r, exact


def main():
    df = pd.read_csv(TRAIN_PATH)
    top_features = load_model_and_top_features(MODEL_PATH, TOP_N)
    candidate_table = build_candidate_table(df["SMILES"])

    print(f"중요도 TOP {TOP_N} Group의 정체 (RDKit 작용기 대조):\n")
    print(f"{'Group':11s} {'중요도':>6s}  {'추정 작용기':18s} {'상관':>7s} {'일치율':>6s}  판정")
    print("-" * 70)

    for group_name, importance in top_features:
        group_values = df[group_name].values
        cand_name, r, exact = match_group(group_values, candidate_table)
        if exact > 0.9:
            verdict = "확실"
        elif abs(r) > 0.7:
            verdict = "유력"
        else:
            verdict = "불명(고차그룹 추정)"
        print(f"{group_name:11s} {importance:6d}  {cand_name:18s} {r:+.3f} {exact*100:5.0f}%  {verdict}")


if __name__ == "__main__":
    main()
```

### RDKit이란?

RDKit은 분자를 다루는 파이썬 라이브러리로써  화학자,연구자들이 쓰는 오픈소스 도구이다.
SMILES 문자열을 진짜 "분자 객체"로 바꿔서, 그 분자의 성질을 계산해준다.

예를 들어 `"CCO"`(에탄올)라는 글자를 주면, RDKit이 이걸 "탄소-탄소-산소로 연결된 분자"로 이해하고

- 분자량이 얼마인지 (`Descriptors.MolWt`)
- -OH가 몇 개인지 (`Fragments.fr_Al_OH`)
- 방향족 고리가 있는지
- 벤젠 고리가 몇 개인지

이런 걸 자동으로 세준다. 


이를 활용하여 RDKit의 결과와 현재 train.csv 파일의 그룹별 개수를 대조하여 역추론이 가능하다.

#### 왜 Group 2가 중요도 1위인가

Group 2(탄소 사슬)가 왜 중요도 1위로 나왔는지 세 가지 가설을 세워 확인하였다.

**가설 A: 값의 범위가 넓어서**  
사슬 길이는 1개부터 40개까지 값이 다양하다. 모델이 데이터를 나눌 때 값의 종류가 많은 피처일수록 나눌 지점이 많아 자주 쓴다. 방향족(0~몇 개)보다 값이 다양해서 더 자주 쓰였을 수 있다.

**가설 B: 거의 모든 분자에 있어서**  
탄소 사슬은 유기분자 대부분에 들어있다. 반면 카복실산은 일부 분자에만 있다. 흔한 피처일수록 여러 분자를 구분하는 데 자주 쓰인다.

**가설 C: 다른 피처와 같이 작용해서**  
사슬 길이가 기본 바탕이 되고, 거기에 어떤 작용기가 붙느냐로 녹는점이 정해진다. 그래서 모델이 사슬로 먼저 나눈 뒤 작용기로 더 나누는 방식일 수 있다.

**확인 결과**

가설 A는 부분적으로 맞았다. Group 2는 값의 종류가 35개, 최댓값 42로 다른 피처보다 범위가 넓다. 자주 쓰일 조건은 맞다. 하지만 이게 전부는 아니었다.

가설 B는 틀렸다. Group 2는 전체 분자의 35%에만 있는데, 오히려 Group 1(53%)과 Group 15(46%)가 더 흔했다. 그러니 "흔해서 1위"는 아니다. 이 가설은 버렸다.

가설 C가 확실히 맞았고, 이게 핵심이다. Group 2를 빼고 학습하면 성능(R²)이 0.533에서 0.457로 떨어진다. 반면 방향족(Group 15)을 빼면 0.036만 떨어진다. 즉 Group 2를 빼면 방향족을 뺄 때보다 2배 이상 성능이 나빠진다. 단순히 자주 쓰인 게 아니라 실제로 중요한 정보를 담고 있다는 뜻이다.

여기에 더해 재미있는 사실도 찾았다. 사슬이 길어질 때 녹는점이 어떻게 변하는지를, 방향족이 있는 분자와 없는 분자로 나눠서 봤다.

- 방향족이 있는 분자: 사슬이 길수록 녹는점이 22.6K 낮아짐
- 방향족이 없는 분자: 사슬이 길수록 녹는점이 1.3K 낮아짐 (거의 변화 없음)

같은 사슬 길이인데 방향족이 있느냐 없느냐에 따라 효과가 약 17배나 차이 났다. 이는 사슬 길이가 혼자 작용하는 게 아니라 방향족과 같이 작용한다는 것을 보여준다. 긴 사슬이 방향족 분자가 촘촘하게 쌓이는 것을 방해해서 녹는점을 크게 낮추는 것으로 보인다.

## 개선 방안

우선 RDKit을 이용하여 SMILES 분자식들에 대한 정보들을 추가로 가져온다.
-> SMILES 문자열을 넣으면 분자량,극성과 같은 숫자 데이터를 받아볼 수 있음

```python
import pandas as pd
import lightgbm as lgb
import joblib
from rdkit import Chem
from rdkit.Chem import Descriptors
from rdkit import RDLogger
from sklearn.model_selection import cross_val_score

RDLogger.DisableLog("rdApp.*")

TRAIN_PATH = "학습 데이터 경로"
MODEL_PATH = "모델 저장 경로"


def get_descriptors(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None
    return {
        "MolWt":         Descriptors.MolWt(mol),
        "HeavyAtom":     Descriptors.HeavyAtomCount(mol),
        "NumHDonors":    Descriptors.NumHDonors(mol),
        "NumHAcceptors": Descriptors.NumHAcceptors(mol),
        "TPSA":          Descriptors.TPSA(mol),
        "MolLogP":       Descriptors.MolLogP(mol),
        "RotBonds":      Descriptors.NumRotatableBonds(mol),
        "AromRings":     Descriptors.NumAromaticRings(mol),
        "FracCSP3":      Descriptors.FractionCSP3(mol),
        "RingCount":     Descriptors.RingCount(mol),
}


def load_data(path):
    df = pd.read_csv(path)
    feats = df["SMILES"].apply(get_descriptors)
    valid = feats.notna()
    X = pd.DataFrame([f for f in feats if f is not None])
    y = df.loc[valid, "Tm"].reset_index(drop=True)
    return X, y, list(X.columns)


def build_model():
    return lgb.LGBMRegressor(
        n_estimators=500,
        learning_rate=0.05,
        num_leaves=31,
        random_state=42,
        verbose=-1,
    )


def evaluate(model, X, y):
    r2 = cross_val_score(model, X, y, cv=5, scoring="r2").mean()
    mae = -cross_val_score(model, X, y, cv=5, scoring="neg_mean_absolute_error").mean()
    return r2, mae


def show_importance(model, feature_names, top_n=10):
    ranking = sorted(zip(feature_names, model.feature_importances_),
                     key=lambda pair: pair[1], reverse=True)
    print(f"\n예측에 가장 영향을 준 descriptor TOP {top_n}:")
    for name, score in ranking[:top_n]:
        print(f"  {name:14s} {score}")


def main():
    X, y, feature_names = load_data(TRAIN_PATH)
    print(f"분자 {len(X)}개, descriptor {len(feature_names)}개")
    print(f"피처: {feature_names}")

    model = build_model()

    r2, mae = evaluate(model, X, y)
    print(f"\n교차검증 성능 -> R² = {r2:.3f}, MAE = {mae:.2f} K")

    model.fit(X, y)
    print("최종 모델 학습 완료")

    joblib.dump({"model": model, "features": feature_names}, MODEL_PATH)
    print(f"모델 저장 -> {MODEL_PATH}")

    show_importance(model, feature_names)


if __name__ == "__main__":
    main()
```

```python
import pandas as pd
import lightgbm as lgb
import joblib
from rdkit import Chem
from rdkit.Chem import Descriptors
from rdkit import RDLogger
from sklearn.model_selection import cross_val_score

RDLogger.DisableLog("rdApp.*")

TRAIN_PATH = "학습 데이터 경로"
MODEL_PATH = "모델 저장 경로"

def get_descriptors(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None
    return {
        "MolWt":         Descriptors.MolWt(mol),
        "HeavyAtom":     Descriptors.HeavyAtomCount(mol),
        "NumHDonors":    Descriptors.NumHDonors(mol),
        "NumHAcceptors": Descriptors.NumHAcceptors(mol),
        "TPSA":          Descriptors.TPSA(mol),
        "MolLogP":       Descriptors.MolLogP(mol),
        "RotBonds":      Descriptors.NumRotatableBonds(mol),
        "AromRings":     Descriptors.NumAromaticRings(mol),
        "FracCSP3":      Descriptors.FractionCSP3(mol),
        "RingCount":     Descriptors.RingCount(mol),
    }


def make_features(df, active_groups):
    desc = pd.DataFrame([get_descriptors(s) for s in df["SMILES"]])
    groups = df[active_groups].reset_index(drop=True)
    return pd.concat([groups, desc], axis=1)


def load_data(path):
    df = pd.read_csv(path)
    group_cols = [c for c in df.columns if c.startswith("Group")]
    active_groups = [c for c in group_cols if df[c].sum() > 0]
    X = make_features(df, active_groups)
    y = df["Tm"]
    return X, y, active_groups, list(X.columns)


def build_model():
    return lgb.LGBMRegressor(
        n_estimators=500,
        learning_rate=0.05,
        num_leaves=31,
        random_state=42,
        verbose=-1,
    )


def main():
    X, y, active_groups, feature_names = load_data(TRAIN_PATH)
    print(f"분자 {len(X)}개")
    print(f"결합 피처 {len(feature_names)}개 = Group {len(active_groups)}개 + descriptor 10개")

    model = build_model()

    r2 = cross_val_score(model, X, y, cv=5, scoring="r2").mean()
    mae = -cross_val_score(model, X, y, cv=5, scoring="neg_mean_absolute_error").mean()
    print(f"\n교차검증 성능 -> R² = {r2:.3f}, MAE = {mae:.2f} K")

    model.fit(X, y)
    print("최종 모델 학습 완료")

    joblib.dump({"model": model, "features": feature_names,
                 "active_groups": active_groups}, MODEL_PATH)
    print(f"모델 저장 -> {MODEL_PATH}")


if __name__ == "__main__":
    main()
```

---

Group 분석에서는 337개의 작용기 피처를 사용했고 MAE 39.36이 나왔다. descriptor 단독으로는 10개의 피처만 썼는데 오히려 38.56으로 더 나은 결과가 나왔다. 피처 개수는 훨씬 적은데 성능은 더 좋았다.

이 차이가 발생한 원인은 두 피처가 담는 정보의 성격이 다르기 때문으로 추정된다. Group은 분자를 잘게 쪼갠 국소적인 작용기 정보이고, descriptor는 분자량이나 극성처럼 분자 전체를 요약한 값이다. 잘게 쪼갠 337개보다 분자 전체를 요약한 10개가 녹는점 예측에는 더 효율적인 정보였던 것으로 보인다.

실제 값과 비교했을 때 두 모델 모두 평균 40K 안팎의 오차가 있었는데, 이는 녹는점이 분자간 힘뿐 아니라 결정 구조 같은 요인에도 영향을 받기 때문에 분자 구조 정보만으로는 완전히 예측하기 어려웠던 것으로 추정된다.

최종적으로 이 두 개를 합쳤더니 MAE가 32.96으로 크게 낮아졌다. Group의 국소 정보와 descriptor의 전체 요약 정보가 서로 다른 부분을 보완해준 결과로, 둘을 합쳤을 때 각각 단독으로 썼을 때보다 성능이 확실히 높아졌다.
