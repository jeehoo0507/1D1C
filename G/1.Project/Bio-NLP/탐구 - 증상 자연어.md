학습&테스트 코드
```python 
import pandas as pd
import torch
import torch.nn as nn
from sklearn.preprocessing import LabelEncoder

# 데이터 로드
train = pd.read_csv("dataset-e/1/Training.csv")
test = pd.read_csv("dataset-e/1/Testing.csv")

for df in (train, test):
    if "Unnamed: 133" in df.columns:
        df.drop(columns=["Unnamed: 133"], inplace=True)

# train, test x, y 만들기
le = LabelEncoder()
X_train = torch.tensor(train.drop(columns=["prognosis"]).values, dtype=torch.float32)
y_train = torch.tensor(le.fit_transform(train["prognosis"]), dtype=torch.long)
X_test = torch.tensor(test.drop(columns=["prognosis"]).values, dtype=torch.float32)
y_test = torch.tensor(le.transform(test["prognosis"]), dtype=torch.long)

n_symptoms = X_train.shape[1]   # 132
n_diseases = len(le.classes_)   # 42

# 모델
model = nn.Sequential(
    nn.Linear(n_symptoms, 64),
    nn.ReLU(),
    nn.Linear(64, n_diseases),
)


loss_fn = nn.CrossEntropyLoss()
optimizer = torch.optim.Adam(model.parameters(), lr=0.01)

# 학습
for epoch in range(1000):
    optimizer.zero_grad()
    out = model(X_train)
    loss = loss_fn(out, y_train)
    loss.backward()
    optimizer.step()
    if (epoch + 1) % 20 == 0:
        print(f"epoch {epoch+1}, loss {loss.item():.4f}")

# 추론 및 평가
model.eval()
with torch.no_grad():
    pred = model(X_test).argmax(dim=1)
    acc = (pred == y_test).float().mean()
    print(f"\ntest: {acc.item():.4f}")
```

## 임베딩 모델
가장 큰 고비는 입력 문장을 정해진 증상들로 변환하는 일이다.
단순히 특정 단어가 등장했을 때 카운팅하게 만든다면 한국어의 특성상 상당히 많은 경우를 예외 처리해야한다.
 ( ex) 머리가 아파요, 머리가 뜨거워요, 찌끈거려요 -> 두통)

이를 해결하기 위해 문장 입력을 벡터로 치환하고 이를 db에 있는 단어 벡터들과 내적하는 과정을 통해 자연어로 입력된 증상에 대해 top-k 형식으로 문장에 들어간 증상이 무엇인지 영어로 나타낼 수 있었다.

데이터셋에 들어갈 칼럼들에 한에 결과를 제한하는 방법을 사용하였다.


```python
from sentence_transformers import SentenceTransformer, util
import torch

# 모델 로드
encoder = SentenceTransformer("paraphrase-multilingual-MiniLM-L12-v2")

# 칼럼 제약 걸기
symptoms = ["headache", "fever", "cough", "skin rash",
            "vomiting", "fatigue", "chest pain", "dizziness"]

# 위에 칼럼 벡터화
symptom_emb = encoder.encode(symptoms, convert_to_tensor=True)

def find_symptoms(text, threshold=0.4, topk=5):
    q = encoder.encode(text, convert_to_tensor=True)
    scores = util.cos_sim(q, symptom_emb)[0]  # 코사인 유사도

    # 정렬
    ranked = sorted(zip(symptoms, scores.tolist()), key=lambda x: -x[1])
    print(f"\n입력: {text}")
    for name, score in ranked[:topk]:
        mark = "[v]" if score > threshold else "[-]"
        print(f"  {mark} {name:12s} {score:.3f}")


find_symptoms(input())

```
